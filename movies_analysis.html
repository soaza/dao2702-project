import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Display in float
pd.set_option('display.float_format', lambda x: '%.3f' % x)

CSV_PATH = 'movies.csv'

df = pd.read_csv(CSV_PATH)


# We filter movies after year 2000
cleaned_df = df[df['year'] >= 2000]
# We drop NaN rows
cleaned_df = cleaned_df.dropna()
